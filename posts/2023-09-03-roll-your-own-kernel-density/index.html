using Random
Random.seed!(2023-09-02);
data = [rand(Beta(20, 3), 900); rand(Beta(4, 10), 100)];
function gaussian_kde(x, data, h)
  norm_dist = Normal()
  scaled_data = (x .- data) ./ h
  kde = 1/h * mean(pdf.(norm_dist, scaled_data))
  return kde
end

function triangular_kde(x, data, h)
  scaled_data = (x .- data) ./ h
  kde = 1/h * mean(max.(1 .- abs.(scaled_data), 0))
  return kde
end
triangular_kde (generic function with 1 method)
function rectangular_kde(x, data, h)
  scaled_data = (x .- data) ./ h
  kde = 1/h * mean(-0.5 .<= scaled_data .<= 0.5)
  return kde
end
rectangular_kde (generic function with 1 method)
function epanechnikov_kde(x, data, h)
  scaled_data = (x .- data) ./ h
  kde = 3/(4h) * mean(max.(1 .- scaled_data.^2, 0))
  return kde
end
